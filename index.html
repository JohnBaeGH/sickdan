식구 실시간 시스템 로그

                

                

                    
                    
                

            


            
            

                

                    
                         조회 실패가 나고 있나요?
                    
                    
                

                

                    
Supabase SQL Editor에 아래 쿼리를 입력해 테이블을 생성해야 조회가 시작됩니다.

                    
-- 1. 식단 데이터 저장 테이블 생성
CREATE TABLE IF NOT EXISTS public.sikgoo_menus (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    menu_date DATE NOT NULL,
    meal_type TEXT NOT NULL,
    food_list TEXT[] NOT NULL,
    raw_ocr_text TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL,
    CONSTRAINT unique_date_meal UNIQUE (menu_date, meal_type)
);

-- 2. 요리별 이미지 캐시 테이블 생성
CREATE TABLE IF NOT EXISTS public.sikgoo_images (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    food_name TEXT UNIQUE NOT NULL,
    image_url TEXT NOT NULL,
    calories INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- 3. 테이블 RLS 권한 완전 해제 (모두 읽고 쓸 수 있도록 허용)
ALTER TABLE public.sikgoo_menus DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.sikgoo_images DISABLE ROW LEVEL SECURITY;
